\begin{figure}[!htbp]
  \centering
  \begin{tikzpicture}[
    every node/.style={font=\small},
    fp/.style={rectangle, rounded corners=2pt, draw=blue!55!black, thick,
               fill=blue!8, minimum width=2.1cm, minimum height=0.7cm, align=center},
    quant/.style={rectangle, rounded corners=2pt, draw=red!60!black, thick,
                  fill=red!10, minimum width=2.1cm, minimum height=0.7cm, align=center},
    prop/.style={rectangle, rounded corners=3pt, draw=orange!70!black, thick,
                 fill=orange!15, minimum width=4.2cm, minimum height=1.2cm, align=center},
    arr/.style={-stealth, thick},
    qarr/.style={-stealth, thick, red!70!black, dashed},
    node distance=0.9cm,
  ]

    % ──── 左栏：attention flow ────
    \node[fp] (q)  at (-5.4, 3.0) {\(q,\ k_i\)};
    \node[fp] (zi) at (-5.4, 2.0) {logits \(z_i = q^\top k_i\)};
    \node[fp] (ai) at (-5.4, 1.0) {\(a_i = \mathrm{softmax}(z)_i\)};
    \node[fp] (o)  at (-5.4, 0.0) {\(o = \sum_i a_i v_i\)};

    \draw[arr] (q.south)  -- (zi.north);
    \draw[arr] (zi.south) -- (ai.north);
    \draw[arr] (ai.south) -- (o.north);

    % 左栏标题
    \node[above=1pt of q, font=\footnotesize\bfseries, blue!55!black]
      (title1) {Attention flow (FP16)};

    % ──── 中栏：量化路径 ────
    \node[quant] (qk)  at (-0.6, 2.5) {\(k_i \to \hat k_i\)};
    \node[quant] (qv)  at (-0.6, 0.5) {\(v_i \to \hat v_i\)};

    % 中栏标题
    \node[above=1pt of qk, font=\footnotesize\bfseries, red!60!black]
      (title2) {Quantization};

    % 从左栏到中栏的量化箭头
    \draw[qarr] (q.east)  to[bend left=10]  node[midway, above, font=\scriptsize, red!70!black]
      {\(Q(\cdot)\)} (qk.west);
    \draw[qarr] (o.west)   -- (o.west);% placeholder
    \draw[qarr, <-] (qv.west) to[bend left=10] (o.east);

    % ──── 右栏：传播分支 ────
    \node[prop] (pk) at (4.2, 2.5) {
      \textbf{K 误差 $\to$ 分布扭曲}\\[2pt]
      \footnotesize \(\Delta a_i\) via softmax
    };
    \node[prop] (pv) at (4.2, 0.5) {
      \textbf{V 误差 $\to$ 聚合偏移}\\[2pt]
      \footnotesize \(\Delta o\) via weighted sum
    };

    % 右栏标题
    \node[above=1pt of pk, font=\footnotesize\bfseries, orange!70!black]
      (title3) {Two propagation paths};

    % 从中栏到右栏
    \draw[arr, red!70!black] (qk.east) -- (pk.west);
    \draw[arr, red!70!black] (qv.east) -- (pv.west);

    % ──── 底部：误差分解公式框 ────
    \node[rectangle, rounded corners=3pt, draw=gray!60, fill=gray!8,
          minimum width=11cm, minimum height=1.3cm, align=center]
      (eq) at (-0.6, -1.6) {
      \textbf{Attention 误差分解：}\\[2pt]
      \(\displaystyle
        \hat{o} - o
        = \underbrace{\textstyle\sum_i (\hat a_i - a_i)\, v_i}_{\text{分布侧项（源自 K 误差）}}
        + \underbrace{\textstyle\sum_i \hat a_i\, (\hat v_i - v_i)}_{\text{聚合侧项（源自 V 误差）}}
      \)
    };

    \draw[arr, gray!60] (pk.south) |- (eq.east);
    \draw[arr, gray!60] (pv.south) |- (eq.east);

  \end{tikzpicture}
  \caption{Attention 误差的两条\emph{耦合}传播路径（telescoping 分解）：$K$ 误差经过 softmax 扭曲 attention 分布，
  $V$ 误差经过加权求和偏移输出。底部公式框给出的是 telescoping 恒等式而非正交因子化——
  聚合侧项 $\sum_i \hat a_i (\hat v_i - v_i)$ 中的 $\hat a_i$ 已经携带 $K$ 扰动信息，
  两条路径在数学上相互耦合、不可完全 isolate。
  尽管路径耦合，\textbf{behavior 保持度}（$a$ 与 $o$ 的联合偏移）仍然同时约束两侧：
  分布 $a$ 的保持度直接限定了第一项；
  分布 $a$ 保持 $\Rightarrow$ $\hat a_i$ 接近 $a_i$ $\Rightarrow$ 第二项里的聚合权重也受控，
  使得 behavior 作为联合对象覆盖两条误差来源。
  单 head 示意，省略时间步与 $\sqrt{d_k}$ 归一化。}
  \label{fig:error-decomposition}
\end{figure}
