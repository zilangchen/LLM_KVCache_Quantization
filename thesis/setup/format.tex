% ============================================================
% format.tex — 章节标题 / 编号 / 行距 / 段落格式
% ============================================================
\usepackage{titlesec}
\usepackage{setspace}

% --- 行距：正文 1.5 倍 ---
\onehalfspacing

% --- 段落首行缩进 2 字符 ---
\setlength{\parindent}{2em}

% --- 章标题：小二号黑体，居中，单倍行距，段前段后各 0.5 行 ---
% "第X章" 格式由 ctexbook 的 chapter/name 控制
\ctexset{
  chapter = {
    format     = \centering\zihao{-2}\heiti\bfseries,
    name       = {第,章},
    number     = \chinese{chapter},
    aftername  = \quad,
    beforeskip = 0.5\baselineskip,
    afterskip  = 0.5\baselineskip,
    fixskip    = true,
  },
  section = {
    format     = \raggedright\zihao{-3}\heiti,
    beforeskip = 0.5\baselineskip,
    afterskip  = 0.5\baselineskip,
    fixskip    = true,
  },
  subsection = {
    format     = \raggedright\zihao{4}\heiti,
    beforeskip = 0.5\baselineskip,
    afterskip  = 0.5\baselineskip,
    fixskip    = true,
  },
  paragraph = {
    format     = \raggedright\zihao{-4}\heiti,
    beforeskip = 0.15\baselineskip,
    afterskip  = 0ex,
    runin      = false,
  },
}

% --- 图表标题格式：五号宋体，按章编号用短横线 ---
\captionsetup{
  font         = {small},          % 五号宋体（去除 sf 避免映射到黑体）
  labelsep     = quad,
  skip         = 6pt,
}
\captionsetup[figure]{
  position     = bottom,
  labelfont    = bf,
}
\captionsetup[table]{
  position     = top,
  labelfont    = bf,
}

% --- 图表按章编号 ---
\counterwithin{figure}{chapter}
\counterwithin{table}{chapter}
\counterwithin{equation}{chapter}
\renewcommand{\thefigure}{\thechapter-\arabic{figure}}
\renewcommand{\thetable}{\thechapter-\arabic{table}}
\renewcommand{\theequation}{\thechapter-\arabic{equation}}

% --- 浮动体布局参数：放宽默认限制以减少整页空白 ---
% 默认 topfraction=0.7 / bottomfraction=0.3 / textfraction=0.2 / floatpagefraction=0.5
% 会导致较大的表格/图独占整页，产生大段空白。以下放宽参数：
\renewcommand{\topfraction}{0.95}       % top 浮动体允许占页高 95%
\renewcommand{\bottomfraction}{0.85}    % bottom 浮动体允许占页高 85%
\renewcommand{\textfraction}{0.05}      % 每页至少保留 5% 正文（其余可为浮动体）
\renewcommand{\floatpagefraction}{0.85} % 纯浮动页需填满 85% 才独占
\setcounter{topnumber}{3}               % 单页最多 3 个 top 浮动体
\setcounter{bottomnumber}{3}            % 单页最多 3 个 bottom 浮动体
\setcounter{totalnumber}{5}             % 单页总浮动体上限 5 个
